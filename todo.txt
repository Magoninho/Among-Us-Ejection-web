render(ctx: CanvasRenderingContext2D) {
		this.angle += 1 * Math.PI / 180;
		ctx.fillStyle = "white";
		for (let p = 0; p < this.positions.length; p++) {
			const position = this.positions[p];
			
			ctx.save(); //  sav
			ctx.translate(position[0], position[1]);
			ctx.rotate(this.angle);
			ctx.fillRect(-this.starSize/2, -this.starSize/2, this.starSize, this.starSize);
			ctx.restore();

			// drawCircle(ctx, position[0], position[1], this.starSize, "white", "white", 0)
		}
	}
